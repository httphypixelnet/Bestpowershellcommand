$wshell = New-Object -ComObject Wscript.Shell
$path =  "C:\Users\"+$env:UserName+"\MC-Launcher.jar"
$javaPath = "C:\Program Files (x86)\PaperCut MF Client\runtime\jre\bin\"

if (-not(Test-Path -Path $path)) {

}

cd $javaPath
./java -jar $path
